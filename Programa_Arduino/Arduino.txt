
// Incluímos la librería para poder controlar el servo
#include <Servo.h>
const int ocupado=11;
const int libre=12;
 
// Declaramos la variable para controlar el servo
Servo servoMotor;
 
void setup() {
  // Iniciamos el monitor serie para mostrar el resultado
  Serial.begin(9600);
 
  // Iniciamos el servo para que empiece a trabajar con el pin 9
  servoMotor.attach(9);
  //led
  pinMode(ocupado,OUTPUT);
  pinMode(libre,OUTPUT);
}
 
void loop() {

  if(Serial.available()> 0){
    int input = Serial.read();
    if(input == '1'){
        // Desplazamos a la posición 0º
           servoMotor.write(0);
       // Esperamos 1 segundo
          delay(1000);
          //digitalWrite(ocupado,HIGH);
      }
      if(input == '2'){
        // Desplazamos a la posición 45º
           servoMotor.write(45);
        // Esperamos 1 segundo
          delay(1000);
        //  digitalWrite(ocupado,HIGH);
      }
      if(input == '3'){
        // Desplazamos a la posición 90º
           servoMotor.write(135);
        // Esperamos 1 segundo
          delay(1000);
        //  digitalWrite(ocupado,HIGH);
      }
      if(input == '4'){
        // Desplazamos a la posición 135º
           servoMotor.write(180);
        // Esperamos 1 segundo
          delay(1000);
         // digitalWrite(ocupado,HIGH);
      }
      if(input == '5'){
        // Desplazamos a la posición 180º
           servoMotor.write(90);
        // Esperamos 1 segundo
          delay(1000);

      }
      if(input == 'o'){
          digitalWrite(ocupado,HIGH);
          digitalWrite(libre,LOW);
      }
      if(input == 'l'){
          digitalWrite(ocupado,LOW);
          digitalWrite(libre,HIGH);
      }

      if(input == 'a'){
           servoMotor.write(90);
          delay(5000);
           servoMotor.write(0);
      }
      if(input == 'b'){
           servoMotor.write(90);
          delay(5000);
           servoMotor.write(45);
      }
      if(input == 'c'){
           servoMotor.write(90);
          delay(5000);
           servoMotor.write(135);
      }
      if(input == 'd'){
           servoMotor.write(90);
          delay(5000);
           servoMotor.write(180);

      }
      if(input == 'r'){
           delay(5000);
           servoMotor.write(90);
      }
    }
 
}
